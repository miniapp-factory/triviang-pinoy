function shuffle<T>(array: T[]): T[] {
  const copy = [...array];
  for (let i = copy.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [copy[i], copy[j]] = [copy[j], copy[i]];
  }
  return copy;
}

"use client";

import { Button } from "@/components/ui/button";
import { Share } from "@/components/share";
import { url } from "@/lib/metadata";

type Animal = "cat" | "dog" | "fox" | "hamster" | "horse";

interface ResultProps {
  answers: (Animal | null)[];
  onRetake: () => void;
}

export default function Result({ answers, onRetake }: ResultProps) {
  const scores: Record<Animal, number> = {
    cat: 0,
    dog: 0,
    fox: 0,
    hamster: 0,
    horse: 0,
  };

  answers.forEach((a) => {
    if (a) scores[a] += 1;
  });

  const best = Object.entries(scores).reduce((a, b) => (b[1] > a[1] ? b : a))[0] as Animal;

  const imageSrc = `/${best}.png`;
  const animalNames: Record<Animal, string> = {
    cat: "Cat",
    dog: "Dog",
    fox: "Fox",
    hamster: "Hamster",
    horse: "Horse",
  };

  const shareText = `I am a ${animalNames[best]}! Take the quiz and find out yours. ${url}`;

  return (
    <div className="max-w-md mx-auto p-4 text-center">
      <h2 className="text-2xl font-bold mb-4">You are a {animalNames[best]}!</h2>
      <img src={imageSrc} alt={animalNames[best]} width={512} height={512} className="mx-auto mb-4" />
      <Share text={shareText} />
      <Button variant="outline" className="mt-4" onClick={onRetake}>
        Retake Quiz
      </Button>
    </div>
  );
}
